Usage Example

How to use the card component in your code

                <!-- Basic Card -->
<aeva-card>
  <div slot="header">
    <h4>Card Title</h4>
  </div>
  <div slot="body">
    <p>Card content goes here</p>
  </div>
  <div slot="footer">
    <aeva-button>Action</aeva-button>
  </div>
</aeva-card>

<!-- Interactive Card -->
<aeva-card interactive id="myCard">
  <div slot="body">
    <p>Click me!</p>
  </div>
</aeva-card>

<!-- JavaScript -->
<script>
  const card = document.getElementById('myCard');
  card.addEventListener('card-click', () => {
    console.log('Card clicked!');
  });
</script>

<!-- Custom Styling -->
<aeva-card 
  padding="lg"
  style="--aeva-card-bg: #f0f9ff; --aeva-card-border-color: #0ea5e9;">
  <div slot="body">
    <p>Customized card</p>
  </div>
</aeva-card>
