// Method 1: Direct Query (Simple & Fast)
const selected = document.querySelector('aeva-radio[name="payment"][checked]');
const value = selected ? selected.value : null;
console.log('Selected:', value);

// Method 2: RadioGroupHelper (Recommended)
import { RadioGroupHelper } from '@aeva/ui';

// Get selected value
const value = RadioGroupHelper.getValue('payment');

// Set selected value
RadioGroupHelper.setValue('payment', 'paypal');

// Check if any selected
if (!RadioGroupHelper.hasSelection('payment')) {
  alert('Please select a payment method');
}

// Get the radio element
const radio = RadioGroupHelper.getRadio('payment');

Animation Details

The page component features smooth animations:

  • Open: Slide in from right (100% → 0%), scale (120% → 100%), fade in (0 → 1)

  <div>Tall slide 1</div>
  <div>Tall slide 2</div>
</aeva-carousel>

<aeva-carousel hide-nav hide-indicators>
  <div>Swipe to navigate</div>
  <div>Another slide</div>
</aeva-carousel>

<!-- JavaScript API -->
<script>
  const carousel = document.querySelector('aeva-carousel');
  
  // Navigate programmatically
  carousel.next();  // Go to next slide
  carousel.prev();  // Go to previous slide
  carousel.goTo(2); // Go to specific slide (0-indexed)
  
  // Listen to slide changes
  carousel.addEventListener('slide-change', (e) => {
    console.log('Active slide:', e.detail.activeIndex);
  });
</script>
